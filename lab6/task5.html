                function sale(fruits){
                    const newFruits = [];
                    for(fruit of fruits){
                        newFruits.push({ 
                            name: fruit.name, 
                            price: fruit.price 
                        });
                    }
        
                    for(newFruit of newFruits){
                        newFruit.price = newFruit.price - (newFruit.price * 20) / 100;
                        newFruit.id = Date.now();
                    }
        
                    return newFruits;
                }
        
                const fruits = [
                    { name: "apple", price: 200 },
                    { name: "orange", price: 300 },
                    { name: "grapes", price: 750 },
                ];
        
                console.log("Початковий масив:");
                console.log(fruits);
        
                console.log("Кінцевий масив:");
                console.log(sale(fruits));
