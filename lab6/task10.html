                function checkBrackets(someFn){
                    let check = [];
            
                    for(char of someFn){
            
                        if(char == '(' || char == '[' || char == '{'){
                            check.push(char);
                        }
            
                        if (char == ')' && check.pop() != '(' || 
                            char == ']' && check.pop() != '[' || 
                            char == '}' && check.pop() != '{') {
            
                            return false;
                        }
                    }
            
                    return check.length === 0;
                }    
            
                console.log("(a+b)-) " + checkBrackets("(a+b)-)"));
                console.log("[a+b]-{} " + checkBrackets("[a+b]-{}"));
                console.log("[a+b}-[} " + checkBrackets("[a+b}-[}"));
